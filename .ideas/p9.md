  subscriptionId: string;

export async function activateSubscription(
  input: ActivateSubscriptionInput
): Promise<ActivateSubscriptionResult> {
  // 1. Load entities
  const subscription = await loadSubscription(input.tenantId, input.subscriptionId);
  const lines = await loadSubscriptionLines(input.tenantId, input.subscriptionId);
  
  // 2. Validate business rules
  const validation = validateSubscriptionLines({ subscription, lines });
  if (!validation.valid) {
    throw new ValidationError(validation.errors.join("; "));
  }
  
  // 3. Check state machine transition
  const canActivate = subscriptionStateMachine.canTransition({
    from: subscription.status,
    to: "active",
    context: { hasLines: lines.length > 0, startDateValid: true }
  });
  if (!canActivate) {
    throw new ValidationError(`Cannot activate subscription in ${subscription.status} status`);
  }
  
  // 4. Compute MRR/ARR
  const mrr = computeMRR({ lines, billingPeriod: template.billingPeriod });
  const arr = mrr.mul(12);
  
  // 5. Update database
  const updated = await db.update(subscriptions)
    .set({
      status: "active",
      mrr: mrr.toString(),
      arr: arr.toString(),
      updatedAt: new Date(),
      updatedBy: input.actorId
    })
    .where(eq(subscriptions.id, input.subscriptionId))
    .returning();
  
  // 6. Log MRR change
  await recordSubscriptionLog({
    subscriptionId: subscription.id,
    eventType: "created",
    oldMrr: new Decimal(0),
    newMrr: mrr,
    actorId: input.actorId
  });
  
  // 7. Record domain event
  await recordDomainEvent({
    tenantId: input.tenantId,
    eventType: "SUBSCRIPTION_ACTIVATED",
    entityType: "subscription",
    entityId: subscription.id,
    payload: { mrr: mrr.toString(), arr: arr.toString() },
    actorId: input.actorId
  });
  
  return { subscription: updated[0], lines, validation };
}

#### **5. MRR/ARR Calculation Logic**

```typescript
export function computeMRR(input: {
  lines: SubscriptionLine[];
  billingPeriod: SubscriptionBillingPeriod;
}): Decimal {
  const lineTotal = input.lines.reduce(
    (sum, line) => sum.plus(line.subtotal),
    new Decimal(0)
  );
  
  // Normalize to monthly
  const periodMultiplier: Record<SubscriptionBillingPeriod, Decimal> = {
    weekly: new Decimal(52).div(12),    // ~4.33
    monthly: new Decimal(1),
    quarterly: new Decimal(1).div(3),   // ~0.333
    yearly: new Decimal(1).div(12)      // ~0.083
  };
  
  return lineTotal.mul(periodMultiplier[input.billingPeriod]).toDecimalPlaces(2);
}

export function computeARR(mrr: Decimal): Decimal {
  return mrr.mul(12).toDecimalPlaces(2);
}
```
